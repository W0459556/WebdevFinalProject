#####CALCULATE LANDSCAPE BILL#####

# 2. input, getting user inputs
houseNum = input("Enter House Number: ")
print("")
propDepth = input("Enter property depth (feet): ")
print("")
propWidth = input("Enter property width (feet): ")
print("")
grassType = input("Enter type of grass (fescue, bentgrass, campus): ")
print("")
treeAmount = input("Enter the number of trees: ")
print("")

# 3. process
# define propSqft
propSqft = int(propDepth) * int(propWidth)

# calculate cost
totCost = 1000 # base cost

if (propSqft>5000):
    totCost += 500 # sqft fee

if ((grassType.lower() == "fescue") or (grassType.lower() == "f")):
    grassCost = 0.05 # price per sqft of fescue

elif ((grassType.lower() == "bentgrass") or (grassType.lower() == "b")):
    grassCost = 0.02 # price per sqft of bentgrass

elif ((grassType.lower() == "campus") or (grassType.lower() == "c")):
    grassCost = 0.01 # price per sqft of campus

else: # if invalid grass input
    grassCost = -1 # prevents error message upon incorrect grass input 
    print("Please input a valid grass type and try again.")

if grassCost > 0: # prevents error message incorrect grass input 
    totCost = totCost + (grassCost * propSqft) # adding cost of grass
    
totCost = totCost + (int(treeAmount) * 100) # adding cost of trees if any

# 4. output
if grassCost > 0: # only works when valid grass type was selected
    # print(f"Total cost for house {houseNum} is: ${round((float(totCost)), 2)}") # setting it to a float then limiting to two decimal places - note, could not get this to make it have 2 decimal places if it did not already have 2 decimal places.
    print("Total cost for house %s is $%.2f" % (houseNum, totCost))

def dataCalc(dataUse, dataPrice):
    return dataUse * dataPrice

dataUse = input("Enter data usage (Mb): ")

if(int(dataUse) <= 200):
    dataCost = 20.00

elif(int(dataUse) <= 500):
    dataCost = (dataCalc((int(dataUse)-200), 0.105))+20

    dataCost = (dataCalc((int(dataUse)-500), 0.110))+51.50

else:
    dataCost = 118.00

# function for determining %
def custSexPercentDeterminant(custAge, perMod1, perMod2, perMod3):
    if(15<=int(custAge)<\25):
        percentMod = perMod1
    elif(25<=int(custAge)<\40):
        percentMod = perMod2
    elif(40<=int(custAge)<\70):
        percentMod = perMod3
    else:
        percentMod = -1 # if i didnt include this it broke, as it would attempt to use the finalPriceCalculator even when not directly called, i think, which breaks when there is no percent mod.
    return percentMod

# function for determining the end price
def finalPriceCalculator(percentMod, carPrice):
    finalPrice = (int(carPrice) * float(percentMod))/12
    return finalPrice
